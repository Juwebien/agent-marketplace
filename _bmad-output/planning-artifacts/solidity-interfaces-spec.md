# Interfaces Solidity pour le Marketplace AI Agent sur Base L2

Les interfaces suivantes sont définies en Solidity 0.8.28. Elles intègrent les contraintes spécifiées, y compris les états pour IMissionEscrow, le staking slashable avec un minimum de 1000 AGNT, les fees de 10% (3% burn + 5% insurance + 2% treasury), la compliance ERC-721 pour IAgentRegistry (en supposant une erreur sur ERC-8004, corrigée vers ERC-721 pour un registre d'agents NFT-like), la protection contre MEV via commit-reveal pour les missions, des events pour les transitions d'état, et des custom errors (documentés dans NatSpec mais définis dans les implémentations). Les interfaces utilisent OpenZeppelin v5 pour les extensions (e.g., IERC20, AccessControl). ReentrancyGuard est à implémenter dans les contrats, sur les fonctions avec transfers. NatSpec est fourni pour chaque fonction.

// SPDX-License-Identifier: MIT
pragma solidity 0.8.28;

import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/access/IAccessControl.sol";

/**
 * @title IAGNTToken
 * @dev Interface pour le token AGNT, un ERC20 avec fonctionnalités de staking et slashing.
 *      Staking minimum : 1000 AGNT. Le slashing est possible pour pénalités.
 *      Intègre des fees avec burn (3% des 10% totaux).
 */
interface IAGNTToken is IERC20, IAccessControl {
    /**
     * @dev Stake un montant de tokens AGNT. Minimum 1000 AGNT.
     * @param amount Montant à staker.
     * Emits Staked event.
     * Reverts with custom error InsufficientStake si amount < 1000.
     */
    function stake(uint256 amount) external;

    /**
     * @dev Unstake les tokens stakés.
     * Emits Unstaked event.
     * Reverts with custom error NoStake si rien n'est staké.
     */
    function unstake() external;

    /**
     * @dev Slash un montant de stake pour une adresse (pénalité).
     * @param account Adresse à slasher.
     * @param amount Montant à slasher.
     * Emits Slashed event.
     * Reverts with custom error InsufficientStake si stake insuffisant.
     * Requiert rôle SLASHER_ROLE.
     */
    function slash(address account, uint256 amount) external;

    /**
     * @dev Burn un montant de tokens (pour fees, 3% des 10%).
     * @param amount Montant à burner.
     * Emits Burned event.
     * Reverts with custom error InsufficientBalance si balance insuffisante.
     */
    function burn(uint256 amount) external;

    /**
     * @dev Retourne le montant staké pour une adresse.
     * @param account Adresse à vérifier.
     * @return Montant staké.
     */
    function stakedBalanceOf(address account) external view returns (uint256);
}

pragma solidity 0.8.28;

import "@openzeppelin/contracts/token/ERC721/IERC721.sol";
import "@openzeppelin/contracts/access/IAccessControl.sol";

/**
 * @title IAgentRegistry
 * @dev Interface pour le registre des agents AI, compliant ERC-721 (NFT pour agents).
 *      Permet l'enregistrement et la gestion d'agents avec staking requis.
 */
interface IAgentRegistry is IERC721, IAccessControl {
    /**
     * @dev Enregistre un nouvel agent AI avec métadonnées.
     * @param agentURI URI des métadonnées de l'agent.
     * @return agentId ID de l'agent minté.
     * Emits AgentRegistered event.
     * Requiert staking minimum de 1000 AGNT via IAGNTToken.
     * Reverts with custom error InsufficientStake si stake < 1000.
     */
    function registerAgent(string calldata agentURI) external returns (uint256 agentId);

    /**
     * @dev Met à jour les métadonnées d'un agent existant.
     * @param agentId ID de l'agent.
     * @param newURI Nouvel URI.
     * Emits AgentUpdated event.
     * Reverts with custom error NotOwner si non propriétaire.
     */
    function updateAgent(uint256 agentId, string calldata newURI) external;

    /**
     * @dev Retourne les détails d'un agent.
     * @param agentId ID de l'agent.
     * @return owner Propriétaire, uri URI des métadonnées.
     */
    function getAgent(uint256 agentId) external view returns (address owner, string memory uri);
}

// SPDX-License-Identifier: MIT
pragma solidity 0.8.28;

import "@openzeppelin/contracts/access/IAccessControl.sol";

/**
 * @title IMissionEscrow
 * @dev Interface pour l'escrow des missions, avec 9 états et protection MEV via commit-reveal.
 *      Fees : 10% (3% burn + 5% insurance + 2% treasury) appliqués sur paiement.
 *      États : CREATED, FUNDED, ACCEPTED, IN_PROGRESS, DELIVERED, COMPLETED, DISPUTED, REFUNDED, CANCELLED.
 *      ReentrancyGuard requis sur fonctions avec transfers.
 */
interface IMissionEscrow is IAccessControl {
    enum MissionState { CREATED, FUNDED, ACCEPTED, IN_PROGRESS, DELIVERED, COMPLETED, DISPUTED, REFUNDED, CANCELLED }

    /**
     * @dev Commit un hash pour créer une mission (étape 1 anti-MEV).
     * @param commitHash Hash des détails de la mission.
     * Emits MissionCommitted event.
     * Reverts with custom error InvalidCommit si hash invalide.
     */
    function commitMission(bytes32 commitHash) external;

    /**
     * @dev Reveal les détails de la mission (étape 2) et la crée.
     * @param missionDetails Détails (description, reward, etc.).
     * @param salt Salt utilisé pour le hash.
     * Transitionne à CREATED.
     * Emits MissionCreated event.
     * Reverts with custom error HashMismatch si hash ne match pas.
     */
    function revealMission(string calldata missionDetails, bytes32 salt) external;

    /**
     * @dev Fund une mission avec un reward.
     * @param missionId ID de la mission.
     * @param amount Montant en AGNT.
     * Transitionne à FUNDED.
     * Emits MissionFunded event.
     * Reverts with custom error InvalidState si pas CREATED.
     */
    function fundMission(uint256 missionId, uint256 amount) external;

    /**
     * @dev Accepte une mission par un agent.
     * @param missionId ID de la mission.
     * @param agentId ID de l'agent.
     * Transitionne à ACCEPTED.
     * Emits MissionAccepted event.
     * Reverts with custom error InvalidState si pas FUNDED.
     */
    function acceptMission(uint256 missionId, uint256 agentId) external;

    /**
     * @dev Marque la mission comme en progrès.
     * @param missionId ID de la mission.
     * Transitionne à IN_PROGRESS.
     * Emits MissionInProgress event.
     * Reverts with custom error InvalidState si pas ACCEPTED.
     */
    function startMission(uint256 missionId) external;

    /**
     * @dev Soumet la livraison de la mission.
     * @param missionId ID de la mission.
     * @param deliveryURI URI de la preuve de livraison.
     * Transitionne à DELIVERED.
     * Emits MissionDelivered event.
     * Reverts with custom error InvalidState si pas IN_PROGRESS.
     */
    function deliverMission(uint256 missionId, string calldata deliveryURI) external;

    /**
     * @dev Complète la mission et paie l'agent (après fees).
     *      Applique 10% fees : 3% burn, 5% insurance, 2% treasury.
     * @param missionId ID de la mission.
     * Transitionne à COMPLETED.
     * Emits MissionCompleted event.
     * Reverts with custom error InvalidState si pas DELIVERED.
     */
    function completeMission(uint256 missionId) external;

    /**
     * @dev Dispute une mission.
     * @param missionId ID de la mission.
     * Transitionne à DISPUTED.
     * Emits MissionDisputed event.
     * Reverts with custom error InvalidState si pas DELIVERED.
     */
    function disputeMission(uint256 missionId) external;

    /**
     * @dev Résout un dispute et rembourse.
     * @param missionId ID de la mission.
     * Transitionne à REFUNDED.
     * Emits MissionRefunded event.
     * Reverts with custom error InvalidState si pas DISPUTED.
     * Requiert rôle ARBITER_ROLE.
     */
    function refundMission(uint256 missionId) external;

    /**
     * @dev Annule une mission.
     * @param missionId ID de la mission.
     * Transitionne à CANCELLED.
     * Emits MissionCancelled event.
     * Reverts with custom error InvalidState si pas avant IN_PROGRESS.
     */
    function cancelMission(uint256 missionId) external;

    /**
     * @dev Retourne l'état d'une mission.
     * @param missionId ID de la mission.
     * @return state État actuel.
     */
    function getMissionState(uint256 missionId) external view returns (MissionState state);
}

// SPDX-License-Identifier: MIT
pragma solidity 0.8.28;

import "@openzeppelin/contracts/access/IAccessControl.sol";

/**
 * @title IReputationOracle
 * @dev Interface pour l'oracle de réputation, intégré avec staking et slashing.
 *      Met à jour la réputation basée sur missions et slashes.
 */
interface IReputationOracle is IAccessControl {
    /**
     * @dev Met à jour la réputation d'un agent après une mission.
     * @param agentId ID de l'agent.
     * @param score Score à ajouter (positif ou négatif).
     * Emits ReputationUpdated event.
     * Reverts with custom error InvalidScore si score invalide.
     * Peut trigger slash si score négatif et stake insuffisant.
     */
    function updateReputation(uint256 agentId, int256 score) external;

    /**
     * @dev Retourne le score de réputation d'un agent.
     * @param agentId ID de l'agent.
     * @return score Score de réputation.
     */
    function getReputation(uint256 agentId) external view returns (int256 score);

    /**
     * @dev Slash la réputation et le stake pour mauvaise conduite.
     * @param agentId ID de l'agent.
     * @param amount Montant à slasher (via IAGNTToken).
     * Emits ReputationSlashed event.
     * Reverts with custom error InsufficientStake si stake < amount.
     * Requiert rôle ORACLE_ROLE.
     */
    function slashReputation(uint256 agentId, uint256 amount) external;
}
